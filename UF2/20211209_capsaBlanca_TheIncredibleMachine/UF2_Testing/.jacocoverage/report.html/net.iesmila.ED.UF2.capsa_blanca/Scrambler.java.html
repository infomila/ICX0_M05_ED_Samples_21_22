package net.iesmila.ED.UF2.capsa_blanca;

/**
 *
 * @author Usuari
 */
public class Scrambler {

    public boolean test(String t) {

        if (t == null || t.length() == 0 || t.length() % 2 != 0) {
            return false;
        }

        for (int i = 0; i < t.length() / 2; i++) {
            if (Character.toUpperCase(t.charAt(i)) !=  t.charAt(t.length() -1 - i) ) {
                return false;
            }
        }
        int count[] = new int[26 + 10];
        for (int i = 0; i < t.length(); i++) {
            char c = t.charAt(i);
            if (c >= 'A' && c <= 'Z') {
                count[c - 'A']++;
            } else if (c >= '0' && c <= '9') {
                count[c - '0' + 'Z' - 'A' + 1]++;
            }
        }
        for (int i = 0; i < count.length; i++) {
            if (count[i] == 0) {
                return false;
            }
        }
        String[] ws = t.split(" ");
        if (ws.length == 1) {
            return false;
        }
        for (int p = 0; p < ws.length; p++) {
            String w = ws[p];
            boolean hasNum=false, hasLett=false;
            for (int i = 0; i < w.length(); i++) {
                char c = Character.toUpperCase(t.charAt(i));
                if ( c >= 'A' && c <= 'Z') {
                   hasLett=true;
                } else if (c >= '0' && c <= '9') {
                    hasNum=true;
                }
            }
            if(!hasNum || !hasLett) return false;
        }

        return true;
    }

}
